// How search works
class CurriculumSearch {
    search(query) {
        const lowerQuery = query.toLowerCase();
        return this.searchIndex
            .filter(item => item.searchText.includes(lowerQuery))
            .slice(0, 10);
    }
    
    selectResult(result) {
        // Auto-populates Grade, Subject, Topic dropdowns
        document.getElementById('grade').value = result.grade;
        document.getElementById('subject').value = result.subject;
        document.getElementById('topic').value = result.topicKey;
    }
}

// Dark mode implementation
function toggleTheme() {
    const currentTheme = document.documentElement.getAttribute('data-theme') || 'light';
    const newTheme = currentTheme === 'light' ? 'dark' : 'light';
    document.documentElement.setAttribute('data-theme', newTheme);
    localStorage.setItem('theme', newTheme);
}
